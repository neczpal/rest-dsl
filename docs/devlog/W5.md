# Week 5

## Progress:

- Big grammar ideation changes
    - Support javalike single line (//) and multiline (/* */) comments
    - meta, model and endpoints are separated
    - Nested blocks - everything is under "api"
    - Support same line Model blocks with comma separation `Pet { id: String, name: String }`
    - Support nested paths example:
        ```
            paths {
                /pet {
                    /:petId {// all endpoints defined under have petId as string in the path param by default
                        ...
                    }
                }
            }
        ```
    - Support anonymous endpoints (under paths):
        ```
            /pet/:petId {
                delete     // empty request body, defaults to return 200 on success with empty response body
                get -> Pet // empty request body, defaults to return 200 with Pet in the response body
                put(body: Pet) -> Pet // Pet request body, defaults to return 200 with Pet in the response body
                post(name: String, age: Int) -> Pet // name and age is a queryParam, defaults to return 200 with Pet in the response body
            }
        ```
    - Support still named endpoints add it just after http method
        ```
            /pet/:petId {
                delete removeOne   // empty request body, defaults to return 200 on success with empty response body
                get -> Pet findOne // empty request body, defaults to return 200 with Pet in the response body
                put createOrUpdate(body: Pet) -> Pet // Pet request body, defaults to return 200 with Pet in the response body
                post updateWithForm(name: String, age: Int) -> {201 -> Pet} // name and age is a queryParam, returns with 201 with Pet in the response body
            }
        ```
    - Support single, multiple typed error handling
        ```
            /pet/:petId {
                delete, error: 404 // in case of error returns 404, response body empty
                get -> Pet, error: 404 -> NotFound // in case of error returns 404, response body is NotFound model
                put (body: Pet) -> Pet, error: {400 -> Error, 403} // in case of 400 error reponse is Error model, in case of 403 error its empty
                post updateWithForm(name: String, age: Int) {
                    response {
                        200 -> Pet
                    }
                    error {
                        400
                        403 -> ApiResponse
                        404 {
                            body: ApiResponse
                        }
                    }
                }
            }
        ```
    - Support more structured endpoint description in case shorthands getting too complex, also can be used later for headers
        ```
            /pet/:petId {
                post updateWithForm(name: String, age: Int) {
                    response {
                        200 -> Pet
                    }
                    error {
                        400
                        403 -> ApiResponse
                        404 {
                            body: ApiResponse
                        }
                    }
                }
            }
        ```
